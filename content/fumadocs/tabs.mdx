## Introduction

Create a tabs to show different package manager command used in projects.
To create such a tabs follow below steps.

1. Add tabsComponents inside mdx-Components.mdx file

```mdx
import defaultMdxComponents from "fumadocs-ui/mdx";
import type { MDXComponents } from "mdx/types";
import * as TabsComponents from "fumadocs-ui/components/tabs";

// use this function to get MDX components, you will need it for rendering MDX
export function getMDXComponents(components?: MDXComponents): MDXComponents {
return {
...defaultMdxComponents,
...TabsComponents,
...components,
};
}
```

2. Now add below snippet - to create a tabs

<Tabs groupId="package-manager" items={["npm", "pnpm", "yaml"]} persist>
  <Tab value="npm">npm</Tab>
  <Tab value="pnpm">pnpm create fumadocs-app</Tab>
  <Tab value="yaml">yaml</Tab>
</Tabs>
